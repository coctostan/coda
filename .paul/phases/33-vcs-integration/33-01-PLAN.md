   ```typescript
   case 'activate': {
     const slug = parsed.remainder.trim();
     if (!slug) {
       ctx.ui.notify('Usage: /coda activate <issue-slug>', 'warning');
       return;
     }
     // Check issue exists
     const issuePath = join(codaRoot, 'issues', slug + '.md');
     if (!existsSync(issuePath)) {
       ctx.ui.notify(`Issue "${slug}" not found at ${issuePath}.`, 'error');
       return;
     }
     // Load current state
     const currentState = loadState(statePath);
     if (currentState?.focus_issue === slug) {
       ctx.ui.notify(`Issue "${slug}" is already focused.`, 'warning');
       return;
     }
     // Read issue to get type and phase
     const issueRecord = codaRead({ record: `issues/${slug}` }, codaRoot);
     const issueType = issueRecord.frontmatter?.issue_type ?? 'feature';
     const issuePhase = issueRecord.frontmatter?.phase ?? 'specify';
     // Update state
     const state = currentState ?? createDefaultState();
     state.focus_issue = slug;
     state.phase = issuePhase;
     state.current_task = null;
     state.completed_tasks = [];
     persistState(state, statePath);
     // Create branch
     const branchResult = createBranch(projectRoot, slug, issueType);
     const branchMsg = branchResult.created
       ? `Branch ${branchResult.branch} created.`
       : `On branch ${branchResult.branch}.`;
     ctx.ui.notify(`Activated issue "${slug}" at phase ${issuePhase}. ${branchMsg}`);
     return;
   }
   ```

3. Add helper function `tryCommitAfterTaskComplete`:
   ```typescript
   function tryCommitAfterTaskComplete(
     params: { record: string; fields: Record<string, unknown> },
     codaRoot: string,
     statePath: string,
     projectRoot: string
   ): void {
     try {
       // Only trigger on task records being marked complete
       if (!params.record.includes('/tasks/') || params.fields.status !== 'complete') return;
       const state = loadState(statePath);
       if (!state || state.phase !== 'build') return;
       // Extract task id and title from the record
       const taskId = extractTaskId(params.record);
       const taskTitle = typeof params.fields.title === 'string' ? params.fields.title : `task-${taskId}`;
       commitTask(projectRoot, taskId, taskTitle);
     } catch {
       // VCS commit is best-effort — don't fail the update
     }
   }
   ```
